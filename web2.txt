<script>
// ===== إدارة الصفحات =====
function showPage(id){
  document.querySelectorAll('.page').forEach(p=>p.classList.remove('active'));
  document.getElementById(id).classList.add('active');
}

// ===== إدارة الثيم =====
function toggleTheme(){
  document.body.classList.toggle('dark');
  localStorage.setItem('theme', document.body.classList.contains('dark') ? 'dark' : 'light');
}
if(localStorage.getItem('theme')==='dark') document.body.classList.add('dark');

// ===== المهام =====
let tasks = JSON.parse(localStorage.getItem('tasks')||'[]');
function renderTasks(){
  let html = '';
  tasks.forEach((t,i)=>{
    html+=`<div class="card">${t} <button onclick="removeTask(${i})">حذف</button></div>`;
  });
  document.getElementById('taskList').innerHTML = html;
}
function addTask(){
  let val = document.getElementById('taskInput').value.trim();
  if(val){ tasks.push(val); document.getElementById('taskInput').value=''; saveTasks(); }
}
function removeTask(i){ tasks.splice(i,1); saveTasks(); }
function saveTasks(){ localStorage.setItem('tasks', JSON.stringify(tasks)); renderTasks(); }
renderTasks();

// ===== الملاحظات =====
let notes = JSON.parse(localStorage.getItem('notes')||'[]');
function renderNotes(){
  let html = '';
  notes.forEach((n,i)=>{
    html+=`<div class="card">${n} <button onclick="removeNote(${i})">حذف</button></div>`;
  });
  document.getElementById('noteList').innerHTML = html;
}
function addNote(){
  let val = document.getElementById('noteInput').value.trim();
  if(val){ notes.push(val); document.getElementById('noteInput').value=''; saveNotes(); }
}
function removeNote(i){ notes.splice(i,1); saveNotes(); }
function saveNotes(){ localStorage.setItem('notes', JSON.stringify(notes)); renderNotes(); }
renderNotes();
</script>
